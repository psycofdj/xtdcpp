282 {
283  while(true)
284  {

286  if (pos == string::npos)
287  {
288  if (! text.empty())
289  {

291  }
292  return tokens;
293  }
294  string pre_text = text.substr(0, pos);
295  if (! pre_text.empty())
296  {
297  tokens.push_back(token_ptr(new TokenText(pre_text)));
298  }
299  text = text.substr(pos+1);
300  if (text.empty())
301  {
302  tokens.push_back(token_ptr(new TokenText("{")));
303  return tokens;
304  }
305 
306  // variable
307  if (text[0] == L'$')
308  {
309  pos = text.find("}");
310  if (pos != string::npos)
311  {
312  tokens.push_back(token_ptr (new TokenVar(text.substr(1, pos-1))));
313  text = text.substr(pos+1);
314  }
315  }
316  // control statement
317  else if (text[0] == L'%')
318  {
319  pos = text.find("}");
320  if (pos != string::npos)
321  {
322  string expression = boost::trim_copy(text.substr(1, pos-2));
323  text = text.substr(pos+1);
324  if (boost::starts_with(expression, "for"))
325  {
326  tokens.push_back(token_ptr (new TokenFor(expression)));
327  }
328  else if (boost::starts_with(expression, "if"))
329  {
330  tokens.push_back(token_ptr (new TokenIf(expression)));
331  }
332  else
333  {
334  tokens.push_back(token_ptr (new TokenEnd(boost::trim_copy(expression))));
335  }
336  }
337  }
338  else
339  {
340  tokens.push_back(token_ptr(new TokenText("{")));
341  }
342  }
